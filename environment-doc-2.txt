create the cluster2200 cluster
1. copy the cluster220 VMs to cass10, cass20, cass30
2. disable cassandra, reconfigure networking, and reboot
3. configure ssh keys until the following is working

[root@ansible ansible]# ansible cluster2200 -m ping

    "ping": "pong"

    "changed": false,
    "ping": "pong"
}
cass10.deltakappa.com | SUCCESS => {
    "changed": false,
    "ping": "pong"

4. reconfigure cassandra nodes for the cluster2200 cluster

*** change the cluster_name

ansible cluster2200 -m shell -a"grep cluster_name {{ cassandra22x_yaml }}"

ansible cluster2200 -m shell -a"sed -i 's/cluster220/cluster2200/g' {{ cassandra22x_yaml }}"

cass30.deltakappa.com | CHANGED | rc=0 >>

cass20.deltakappa.com | CHANGED | rc=0 >>

cass30.deltakappa.com | CHANGED | rc=0 >>
cluster_name: 'cluster2200'

cass10.deltakappa.com | CHANGED | rc=0 >>

*** change the seed nodes

ansible cluster2200 -m shell -a"grep seeds\: {{ cassandra22x_yaml }}"

ansible cluster2200 -m shell -a"sed -i 's/cass1.deltakappa.com,cass2.deltakappa.com,cass3.deltakappa.com/cass10.deltakappa.com,cass20.deltakappa.com,cass30.deltakappa.com/g' {{ cassandra22x_yaml }}"

          - seeds: "cass10.deltakappa.com,cass20.deltakappa.com,cass30.deltakappa.com"

          - seeds: "cass10.deltakappa.com,cass20.deltakappa.com,cass30.deltakappa.com"

cass30.deltakappa.com | CHANGED | rc=0 >>
          - seeds: "cass10.deltakappa.com,cass20.deltakappa.com,cass30.deltakappa.com"

*** change the listen & rpc addresses

ansible cluster2200 -m shell -a"grep listen_address\:cass {{ cassandra22x_yaml }}"
ansible cluster2200 -m shell -a"grep rpc_address\:cass {{ cassandra22x_yaml }}"

ansible cluster2200 -m shell -a"sed -i 's/cass1.deltakappa.com/cass10.deltakappa.com/g' {{ cassandra22x_yaml }}"
ansible cluster2200 -m shell -a"sed -i 's/cass2.deltakappa.com/cass20.deltakappa.com/g' {{ cassandra22x_yaml }}"
ansible cluster2200 -m shell -a"sed -i 's/cass3.deltakappa.com/cass30.deltakappa.com/g' {{ cassandra22x_yaml }}"

ansible cluster2200 -m shell -a"grep 'listen_address\: cass' {{ cassandra22x_yaml }}"
ansible cluster2200 -m shell -a"grep 'rpc_address\: cass' {{ cassandra22x_yaml }}"

*** reset the node log, commit log, cache, and data

ansible cluster2200 -m shell -a"rm -fR {{ cassandra22x_var_path }}/archive/*"
ansible cluster2200 -m shell -a"rm -f {{ cassandra22x_var_path }}/commitlog/*"
ansible cluster2200 -m shell -a"rm -f {{ cassandra22x_var_path }}/saved_caches/*"
ansible cluster2200 -m shell -a"rm -f {{ cassandra22x_log_path }}/*"
ansible cluster2200 -m shell -a"rm -fR {{ cassandra22x_data_path }}/*"

enable cassandra and start
ansible cluster2200 -m shell -a"systemctl enable cassandra"
ansible cluster2200 -m shell -a"systemctl start cassandra"
